\chapter{Bonus Proofs}

\section{\texorpdfstring{$\text{ZFC}+\text{MA}\vdash\mf{dd}=\mf{c}$}{MA=>dd=c}}

We already knew that MA implies $\mf{dd}=\mf{c}$, since, as we saw, $\text{non}(\mc{M})\leq\mf{dd}$, and we know MA collapses CichoĹ„'s Diagram. However, let's do a direct proof of it by exhibiting a suitable poset.

\begin{theorem}{}{}
    $\text{ZFC}+\text{MA}(\kappa)\vdash\mf{dd}>\kappa$.
\end{theorem}

\section{Take 1}

Let $S\subseteq\text{Sym}(\omega)$, with $\lvert S\rvert\leq\kappa$, and let's assume $\text{MA}(\kappa)$. Now, consider

$$\mbb{P}=\{(s,P,k):s\in2^{<\omega},\;\text{id}\in P\in[S]^{<\omega}\;k>0\}$$

Intuitively, $s\in 2^{<\omega}$ is a finite fragment of a characteristic function of the set we want to construct; $P\in [S]^{<\omega}$ is a finite set of permutations conditions we want to assure, $k>0$ will be our error $\frac1k$ between finite fragments of $A$ and $p[A]$. We will denote by $A_s$ (or just $A$), the set $\{n\in\omega:s(n)=1\}$.

Also, we considered $\text{id}\in P$ in the definition of $\mbb{P}$ just for simplicity, since we will eventually want to guarantee that $d(A)$ exists and $d(A)=d(p[A])$, we can already let $p=\text{id}$ and use the same Cauchy's Criterion as we will see next in the definition of $\leq$:

Let $p_1=(s_1,P_1,k_1)\leq (s_2,P_2,k_2)=p_2$ if, and only if, $s_1\supseteq s_2$, $P_1\supseteq P_2$, $k_1\geq k_2$ and, for all $n,m>0$ such that $\text{dom}(s_2)\leq n,m\leq\text{dom}(s_1)$, and all $p\in P_2$, we have
$$\lvert d_n(A_{s_1})-d_m(p[A_{s_1}])\rvert\leq\frac1{k_2}-\frac1{k_1}$$

We used $\frac1{k_2}-\frac1{k_1}$, instead of just $\frac1{k_2}$, to assure we can do a triangle inequality trick in order to prove $(\mbb{P},\leq)$ is a poset.

The idea is to consider the following (expected dense) sets, for each $n\in\omega$ and $p\in P$
\begin{align*}
    C_n=\{(s,P,k)\in\mbb{P}:k\geq n\}&\quad\text{and}\quad E_p=\{(s,P,k)\in\mbb{P}:p\in P\}\\
    D_n=\{(s,P,k)\in\mbb{P}:\exists m\geq n\;[s(m)=1]\}&\quad\text{and}\quad \widetilde{D}_n=\{(s,P,k)\in\mbb{P}:\exists m\geq n\;[s(m)=0]\}
\end{align*}
$$F_n = \{(s,P,k):n\in\text{dom}(s)\}$$

The set $F_n$ is, as in the Cohen/Easton Forcing, to guarantee that our generic object $s_G$ will have domain $\omega$. $D_n$ and $\widetilde{D}_n$ are to guarantee $s$ is the characteristic function of an infinite and coinfinite set. $E_p$ is to assure we can prove the condition for all $p\in P$ and $C_n$ to guarantee we can let $\frac1k$ be a decreasing sequence.

Then, given $G$ our $\mc{D}$-generic filter, with $\mc{D}=\{E_p,C_n,D_n,\widetilde{D}_n,F_n:n\in\omega,\,p\in S\}$, let
$$s_G=\bigcup\{s:\exists P\;\exists k\;[(s,P,k)\in G]\}$$

\begin{proof}
    For now on, let $N_i:=\text{dom}(s_i)$ and $N=\text{dom}(s)$. Let's start proving $(\mbb{P},\leq)$ is in fact a poset:
    \begin{itemize}
        \item \textbf{Antisymmetry:} Trivial
        \item \textbf{Reflexivity:} \textcolor{laser_eyes}{Also wrong, bruh} if $N\leq n,m\leq N$, then $n=m=N$ and, for all $p\in P$, since $A_s$ is a finite set, we will have that $\lvert A\rvert=\lvert p[A]\rvert$, so
        $$0=\lvert d_N(A)-d_N(p[A])\rvert\leq\frac1k-\frac1k=0$$
        \item \textbf{Transitivy:} Let $p_i=(s_i,P_i,k_i)$, for $i=1,2,3$, where $p_1\leq p_2\leq p_3$. Clearly $s_1\supseteq s_3$, $P_1\supseteq P_3$ and $k_1\geq k_3$. Now, given $n,m>0$ such that $N_3\leq n,m\leq N_1$. If either $n,m\leq N_2$ or $N_2\leq n,m$, we can clearly just apply $p_2\leq p_3$ or $p_1\leq p_2$, respectively, to guarantee our condition for all $p\in P_3\subseteq P_2,P_1$.

        Assume then that $N_3\leq n\leq N_2\leq m\leq N_1$, then we have that, for all $p\in P_3$
        \begin{align*}
           \lvert d_n(A_{s_2})-d_{N_2}(A_{s_2})\rvert\leq\frac1{k_3}-\frac1{k_2}\tag{$\text{id}\in P_3$}\\
            \lvert d_{N_2}(A_{s_1})-d_m(p[A_{s_1}])\rvert\leq\frac1{k_2}-\frac1{k_1}
        \end{align*}
        summing, using $A_{s_2}\cap k=A_{s_1}\cap k$, for $k\leq N_2$, and triangle inequality, we conclude that
        $$\lvert d_n(A_{s_1})-d_m(p[A_{s_1}])\rvert\leq\frac1{k_3}-\frac1{k_1}$$
    \end{itemize}
    Let's now show that $\mc{D}$ is in fact a family of dense sets:
    \begin{itemize}
        \item $C_n$: Trivial, since $(s,P,k)\leq (s,P,m)$, for all $m\geq k$;

        \item \textcolor{laser_eyes}{$D_n$, $\widetilde{D}_n$ and $F_n$: I don't know how to guarantee they are dense;}
        \item $E_p$: Similarly, it follows from $(s,P\cup\{p\},m)\leq(s,P,m)$.
    \end{itemize}
    Now, since $\lvert\mc{D}\rvert\leq\lvert S\rvert\leq\kappa$, we can apply $\text{MA}(\kappa)$ to find our $\mc{D}$-generic filter $G$ and define $s_G$. $F_n$ guarantees us that $s_G\in 2^\omega$; both $D_n$ and $\widetilde{D}_n$ guarantees $A_{s_G}$ is infinite and coinfinite.

    Now, fix $p\in S$, $\varepsilon>0$ and let $\frac1n<\varepsilon$. Since $G$ is a $\mc{D}$-generic filter, we can find $p_1\in G\cap C_n$, $p_2\in G\cap E_p$ and a $p_3\leq p_1,p_2$. We claim that $N=\text{dom}(s_3)$ is our initial point to verify Cauchy's Criterion for $\varepsilon$. Let then $n,m\geq N$, we will show $\lvert d_n(A_{s_G})-d_m(A_{s_G})\rvert<\varepsilon$.

    Let $M=\max(p^{-1}[\max\{m,n\}])$, then we can find $p_4\in G\cap F_M$, let $G\ni q\leq p_3,p_4$. Then, by now, $q=(s,P,k)$ would be such that $p\in P$, $\frac1k<\varepsilon$ and $M\subseteq \text{dom}(s)$. Since $q\leq p_3$, we have
    $$\lvert d_n(A_s)-d_m(p[A_s])\rvert<\varepsilon$$
    Now, we just need to prove that $p[A_s]\cap m=p[A_{s_G}]\cap m$. The $(\subseteq)$ is clear. Now, let $\ell \in p[A_{s_G}]\cap m$, then $\ell\leq m$, but $p^{-1}[\max\{m,n\}]\subseteq M\subseteq\text{dom}(s)$, so $p^{-1}(\ell)\in\text{dom}(s)$. Now, since $p^{-1}(\ell)\in A_{s_G}$, then $p^{-1}(\ell)\in A_s$ too, implying $\ell\in p[A_s]\cap m$, as desired.
\end{proof}

\section{Take 3}

Let
$$\mbb{P}=\{p=(s_p,F_p, \varepsilon_p):s_p\in2^{<\omega},\text{id}\in F_p\in[\mc{F}]^{<\omega},\varepsilon\in\mbb{Q}_+\}$$
Also, let $K(p,\pi)=\max\{k\in\omega:\pi^{-1}[k]\subseteq\text{dom}(s_p)\}$, and $A_p=A_{s_p}$. Then define $p\leq q$ as
$$s_p\supseteq s_q, F_p\supseteq F_q,\varepsilon_p\leq\varepsilon_q,\;\lvert d_n(\pi[A_p])-1/2\rvert<\varepsilon_q,\forall n\in K(p,\pi)\setminus K(q,\pi),\forall\pi\in F_q$$
this is clearly reflexive and symmetric. For transitiveness, let $p\leq q\leq r$, then
\begin{itemize}
    \item If $n\in K(p,\pi)\setminus K(q,\pi)$, for $\pi\in F_q\supseteq F_r$ we have
    $$\lvert d_n(\pi[A_p])-1/2\rvert<\varepsilon_q\leq\varepsilon_r$$
    \item If $n\in K(q,\pi)\setminus K(r,\pi)$, for $\pi\in F_r$, then, since $\pi^{-1}[n]\subseteq A_q\subseteq_c A_p$ we have, as in the previous try, that $n\cap \pi[A_q]=n\cap\pi[A_p]$, and so $\lvert d_n(\pi[A_p])-1/2\rvert<\varepsilon_r$.
\end{itemize}

So $(\mbb{P},\leq)$ is a poset that, as before, is clearly ccc, since it's $\sigma$-centered. Now, consider the sets
\begin{itemize}
    \item $D_\varepsilon=\{p\in\mbb{P}:\varepsilon_p<\varepsilon\}$, this is trivially dense;
    \item $D_\pi=\{p\in\mbb{P}:\pi\in F_p\}$, also trivially dense;
    \item $D_n=\{p\in\mbb{P}:\exists k\geq m\geq n\text{ such that }s_p(m)=0\text{ and }s_p(k)=1\}$;
    \item $E_n=\{p\in\mbb{P}:n\subseteq\text{dom}(s_p)\}$.
\end{itemize}
